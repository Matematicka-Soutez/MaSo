const logger = require('./../logger').errorLogger
 
class ResponseError extends Error {
  constructor(message, type, status) {
    super()
    Error.captureStackTrace(this, this.constructor)
    this.name = this.constructor.name
    this.message = message
    this.type = type
    this.status = status
    logger.error({
      error: {
        name: this.name,
        message,
        type,
        status,
      },
    })
  }
}

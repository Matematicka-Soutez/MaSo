module.exports = (sequelize, DataTypes) => {
  const Hobby = sequelize.define('Hobby', {
    name: { type: DataTypes.DATE, field: 'name' },
  }, {
    tableName: 'Hobbies',
  })
 
  Hobby.associate = models => {
    Hobby.belongsToMany(models.User, {
      as: 'users',
      foreignKey: { name: 'hobbyId', field: 'hobby_id' },
      through: 'user_hobbies',
      onDelete: 'RESTRICT',
    })
    Hobby.belongsTo(models.User, {
      as: 'createdBy',
      foreignKey: { name: 'userId', field: 'user_id' },
      onDelete: 'RESTRICT',
    })
    Hobby.belongsTo(models.RecordImportanceType, {
      as: 'recordImportanceType',
      foreignKey: { name: 'recordImportanceTypeId', field: 'record_importance_type_id' },
      onDelete: 'RESTRICT',
    })
  }
 
  return Hobby
}

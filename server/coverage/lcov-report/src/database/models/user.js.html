module.exports = (sequelize, DataTypes) => {
  const User = sequelize.define('User', {
    /* REGISTRATION RELATED PROPERTIES */
    // Step 1
    email: { type: DataTypes.STRING, allowNull: false },
    registrationStep: { type: DataTypes.INTEGER, allowNull: false, defaultValue: 2, field: 'registration_step' },
    // Step 2
    firstName: { type: DataTypes.STRING, allowNull: true, field: 'first_name' },
    lastName: { type: DataTypes.STRING, allowNull: true, field: 'last_name' },
    dob: { type: DataTypes.DATE, allowNull: true, field: 'dob' },
    // Step 4
    weight: { type: DataTypes.DOUBLE, allowNull: true, field: 'weight' },
    height: { type: DataTypes.DOUBLE, allowNull: true, field: 'height' },
    // Step 10
    password: { type: DataTypes.STRING, allowNull: true },
    /* ADMINISTRATIVE PROPERTIES */
    publicToken: { type: DataTypes.STRING, field: 'public_token' },
    passwordPublicToken: { type: DataTypes.STRING, field: 'password_public_token' },
    duplicateResetPasswordToken: { type: DataTypes.STRING, field: 'duplicate_reset_password_token' },
    confirmed: { type: DataTypes.BOOLEAN, allowNull: false, defaultValue: false, field: 'confirmed' },
    passwordLastUpdatedAt: { type: DataTypes.DATE, defaultValue: new Date(), field: 'password_last_updated_at' },
    lastLoginAt: { type: DataTypes.DATE, field: 'last_login_at' },
  }, {
    tableName: 'Users',
    timestamps: true,
  })
 
  User.associate = models => {
    User.belongsToMany(models.Hobby, {
      as: 'hobbies',
      foreignKey: { name: 'userId', field: 'user_id' },
      through: 'user_hobbies',
      onDelete: 'RESTRICT',
    })
 
    User.belongsTo(models.Address, {
      as: 'address',
      foreignKey: { name: 'addressId', field: 'address_id' },
      onDelete: 'RESTRICT',
    })
 
    User.belongsTo(models.WeeklyExcerciseFrequency, {
      as: 'weeklyExcerciseFrequency',
      foreignKey: { name: 'weeklyExcerciseFrequencyId', field: 'weekly_excercise_frequency_id' },
      onDelete: 'RESTRICT',
    })
 
    User.belongsTo(models.Gender, {
      as: 'gender',
      foreignKey: { name: 'genderId', field: 'gender_id' },
      onDelete: 'RESTRICT',
    })
  }
 
  return User
}
